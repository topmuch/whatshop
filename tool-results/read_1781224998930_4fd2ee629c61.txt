   450→   450→  function downloadQrCode() {
   451→   451→    if (!qrDataUrl || !shop) return
   452→   452→    const img = new Image()
   453→   453→    img.crossOrigin = 'anonymous'
   454→   454→    img.onload = () => {
   455→   455→      const canvas = document.createElement('canvas')
   456→   456→      canvas.width = 800
   457→   457→      canvas.height = 900
   458→   458→      const ctx = canvas.getContext('2d')
   459→   459→      if (!ctx) return
   460→   460→      ctx.fillStyle = '#FFFFFF'
   461→   461→      ctx.fillRect(0, 0, 800, 900)
   462→   462→      const qrSize = 600
   463→   463→      const offsetX = (800 - qrSize) / 2
   464→   464→      ctx.drawImage(img, offsetX, 60, qrSize, qrSize)
   465→   465→      ctx.fillStyle = '#1a1a2e'
   466→   466→      ctx.font = 'bold 28px system-ui, -apple-system, sans-serif'
   467→   467→      ctx.textAlign = 'center'
